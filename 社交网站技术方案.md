# 小众兴趣爱好者社交网站 - 技术方案

## 项目概述

### 项目背景
为小众兴趣爱好者打造一个专属的社交平台，让用户能够找到志同道合的伙伴，分享兴趣内容，建立深度连接。

### 目标用户
- 小众兴趣爱好者（如：手账、模型、独立音乐、小众运动等）
- 预计用户规模：100-1000人（MVP阶段）

### 核心价值
- 精准的兴趣匹配和内容推荐
- 高质量的社区氛围
- 深度交流而非浅层社交

## 功能模块设计

### 模块一：用户管理

#### 功能列表
- 用户注册（邮箱/手机号）
- 用户登录（密码/验证码）
- 个人信息管理
- 兴趣标签设置
- 账号安全设置

#### 数据模型
```sql
-- 用户表
CREATE TABLE users (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    username VARCHAR(50) UNIQUE NOT NULL,
    email VARCHAR(100) UNIQUE NOT NULL,
    phone VARCHAR(20) UNIQUE,
    password_hash VARCHAR(255) NOT NULL,
    nickname VARCHAR(50),
    avatar VARCHAR(255),
    bio TEXT,
    status TINYINT DEFAULT 1 COMMENT '0:禁用 1:正常 2:待验证',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    INDEX idx_email (email),
    INDEX idx_phone (phone)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 用户兴趣标签表
CREATE TABLE user_interests (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    user_id BIGINT NOT NULL,
    tag_id BIGINT NOT NULL,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id),
    FOREIGN KEY (tag_id) REFERENCES interest_tags(id),
    UNIQUE KEY uk_user_tag (user_id, tag_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 兴趣标签表
CREATE TABLE interest_tags (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(50) NOT NULL,
    category VARCHAR(50),
    icon VARCHAR(255),
    sort_order INT DEFAULT 0,
    is_active TINYINT DEFAULT 1
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
```

### 模块二：个人主页

#### 功能列表
- 个人资料展示
- 动态时间线
- 关注/粉丝列表
- 个人成就/徽章
- 访客记录

#### 数据模型
```sql
-- 关注关系表
CREATE TABLE follows (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    follower_id BIGINT NOT NULL COMMENT '关注者ID',
    following_id BIGINT NOT NULL COMMENT '被关注者ID',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (follower_id) REFERENCES users(id),
    FOREIGN KEY (following_id) REFERENCES users(id),
    UNIQUE KEY uk_follow (follower_id, following_id),
    INDEX idx_following (following_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
```

### 模块三：动态系统

#### 功能列表
- 发布动态（文字+图片）
- 动态点赞
- 评论系统
- 动态收藏
- 内容举报
- 话题标签

#### 数据模型
```sql
-- 动态表
CREATE TABLE posts (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    user_id BIGINT NOT NULL,
    content TEXT NOT NULL,
    images JSON COMMENT '图片URL数组',
    topic_id BIGINT,
    like_count INT DEFAULT 0,
    comment_count INT DEFAULT 0,
    status TINYINT DEFAULT 1 COMMENT '0:删除 1:正常 2:审核中',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id),
    INDEX idx_user_created (user_id, created_at),
    INDEX idx_topic (topic_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 点赞表
CREATE TABLE likes (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    user_id BIGINT NOT NULL,
    post_id BIGINT NOT NULL,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id),
    FOREIGN KEY (post_id) REFERENCES posts(id),
    UNIQUE KEY uk_user_post (user_id, post_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 评论表
CREATE TABLE comments (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    user_id BIGINT NOT NULL,
    post_id BIGINT NOT NULL,
    parent_id BIGINT COMMENT '父评论ID，用于回复',
    content TEXT NOT NULL,
    like_count INT DEFAULT 0,
    status TINYINT DEFAULT 1,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id),
    FOREIGN KEY (post_id) REFERENCES posts(id),
    FOREIGN KEY (parent_id) REFERENCES comments(id),
    INDEX idx_post_created (post_id, created_at)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 话题表
CREATE TABLE topics (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(100) NOT NULL,
    description TEXT,
    cover VARCHAR(255),
    post_count INT DEFAULT 0,
    is_active TINYINT DEFAULT 1,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
```

### 模块四：私信系统

#### 功能列表
- 私信聊天（一对一）
- 消息已读状态
- 历史消息记录
- 消息通知
- 图片/文件发送

#### 数据模型
```sql
-- 会话表
CREATE TABLE conversations (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    user1_id BIGINT NOT NULL,
    user2_id BIGINT NOT NULL,
    last_message_id BIGINT,
    last_message_at DATETIME,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user1_id) REFERENCES users(id),
    FOREIGN KEY (user2_id) REFERENCES users(id),
    UNIQUE KEY uk_users (user1_id, user2_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 私信表
CREATE TABLE messages (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    conversation_id BIGINT NOT NULL,
    sender_id BIGINT NOT NULL,
    content TEXT,
    message_type TINYINT DEFAULT 1 COMMENT '1:文字 2:图片 3:文件',
    media_url VARCHAR(255),
    is_read TINYINT DEFAULT 0,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (conversation_id) REFERENCES conversations(id),
    FOREIGN KEY (sender_id) REFERENCES users(id),
    INDEX idx_conversation (conversation_id, created_at)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
```

### 模块五：群组系统

#### 功能列表
- 创建群组
- 加入/退出群组
- 群组动态
- 群组公告
- 群组管理（管理员权限）
- 群组搜索

#### 数据模型
```sql
-- 群组表
CREATE TABLE groups (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(100) NOT NULL,
    description TEXT,
    avatar VARCHAR(255),
    cover VARCHAR(255),
    category VARCHAR(50),
    creator_id BIGINT NOT NULL,
    member_count INT DEFAULT 1,
    is_public TINYINT DEFAULT 1 COMMENT '0:私密 1:公开',
    status TINYINT DEFAULT 1,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (creator_id) REFERENCES users(id),
    INDEX idx_category (category)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 群组成员表
CREATE TABLE group_members (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    group_id BIGINT NOT NULL,
    user_id BIGINT NOT NULL,
    role TINYINT DEFAULT 1 COMMENT '1:成员 2:管理员 3:群主',
    joined_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (group_id) REFERENCES groups(id),
    FOREIGN KEY (user_id) REFERENCES users(id),
    UNIQUE KEY uk_group_user (group_id, user_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 群组动态表
CREATE TABLE group_posts (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    group_id BIGINT NOT NULL,
    user_id BIGINT NOT NULL,
    post_id BIGINT NOT NULL,
    is_pinned TINYINT DEFAULT 0,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (group_id) REFERENCES groups(id),
    FOREIGN KEY (user_id) REFERENCES users(id),
    FOREIGN KEY (post_id) REFERENCES posts(id),
    INDEX idx_group_created (group_id, created_at)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 群组公告表
CREATE TABLE group_announcements (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    group_id BIGINT NOT NULL,
    title VARCHAR(200) NOT NULL,
    content TEXT NOT NULL,
    creator_id BIGINT NOT NULL,
    is_active TINYINT DEFAULT 1,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (group_id) REFERENCES groups(id),
    FOREIGN KEY (creator_id) REFERENCES users(id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
```

## API 接口设计

### 认证相关
```
POST   /api/v1/auth/register        # 用户注册
POST   /api/v1/auth/login           # 用户登录
POST   /api/v1/auth/logout          # 用户登出
POST   /api/v1/auth/refresh         # 刷新令牌
POST   /api/v1/auth/forgot-password # 忘记密码
POST   /api/v1/auth/reset-password  # 重置密码
```

### 用户相关
```
GET    /api/v1/users/me             # 获取当前用户信息
PUT    /api/v1/users/me             # 更新当前用户信息
GET    /api/v1/users/{id}           # 获取指定用户信息
GET    /api/v1/users/{id}/posts     # 获取用户动态列表
POST   /api/v1/users/{id}/follow    # 关注用户
DELETE /api/v1/users/{id}/follow    # 取消关注
GET    /api/v1/users/{id}/followers # 获取粉丝列表
GET    /api/v1/users/{id}/following # 获取关注列表
```

### 动态相关
```
GET    /api/v1/posts                # 获取动态列表（首页信息流）
POST   /api/v1/posts                # 发布动态
GET    /api/v1/posts/{id}           # 获取动态详情
PUT    /api/v1/posts/{id}           # 更新动态
DELETE /api/v1/posts/{id}           # 删除动态
POST   /api/v1/posts/{id}/like      # 点赞动态
DELETE /api/v1/posts/{id}/like      # 取消点赞
GET    /api/v1/posts/{id}/comments  # 获取评论列表
POST   /api/v1/posts/{id}/comments  # 发表评论
```

### 私信相关
```
GET    /api/v1/conversations        # 获取会话列表
GET    /api/v1/conversations/{id}   # 获取会话详情

POST   /api/v1/conversations/{id}/messages  # 发送消息
```

### 群组相关

## 部署方案

### 开发环境
```bash
# 后端启动
cd backend
python -m venv venv
source venv/bin/activate  # Windows: venv\Scripts\activate
pip install -r requirements.txt
uvicorn app.main:app --reload --host 0.0.0.0 --port 8000

# 前端启动
cd frontend
npm install
npm run dev
```

### Docker 部署
```yaml
# docker-compose.yml
version: '3.8'

services:
  # MySQL 数据库
  mysql:
    image: mysql:8.0
    container_name: social-mysql
    environment:
      MYSQL_ROOT_PASSWORD: rootpassword
      MYSQL_DATABASE: social_network
      MYSQL_USER: social_user
      MYSQL_PASSWORD: social_password
    ports:
      - "3306:3306"
    volumes:
      - mysql_data:/var/lib/mysql
    networks:
      - social-network

  # Redis 缓存
  redis:
    image: redis:7-alpine
    container_name: social-redis
    ports:
      - "6379:6379"
    networks:
      - social-network

  # 后端服务
  backend:
    build: ./backend
    container_name: social-backend
    environment:
      - DATABASE_URL=mysql+asyncysql://social_user:social_password@mysql:3306/social_network
      - REDIS_URL=redis://redis:6379/0
      - SECRET_KEY=your-secret-key-here
    ports:
      - "8000:8000"
    depends_on:
      - mysql
      - redis
    networks:
      - social-network

  # 前端服务
  frontend:
    build: ./frontend
    container_name: social-frontend
    ports:
      - "80:80"
    depends_on:
      - backend
    networks:
      - social-network

  # Nginx 反向代理
  nginx:
    image: nginx:alpine
    container_name: social-nginx
    ports:
      - "80:80"
      - "443:443"
    volumes:
      - ./nginx/nginx.conf:/etc/nginx/nginx.conf
      - ./nginx/ssl:/etc/nginx/ssl
    depends_on:
      - backend
      - frontend
    networks:
      - social-network

volumes:
  mysql_data:

networks:
  social-network:
    driver: bridge
```

### 环境变量配置
```bash
# .env 文件示例

# 数据库配置
DATABASE_URL=mysql+asyncysql://user:password@localhost:3306/social_network
DATABASE_ECHO=False

# Redis 配置
REDIS_URL=redis://localhost:6379/0

# JWT 配置
SECRET_KEY=your-secret-key-here
ACCESS_TOKEN_EXPIRE_MINUTES=30
REFRESH_TOKEN_EXPIRE_DAYS=7

# 文件上传配置
UPLOAD_DIR=./uploads
MAX_FILE_SIZE=10485760  # 10MB

# 邮件配置（可选）
SMTP_HOST=smtp.example.com
SMTP_PORT=587
SMTP_USER=your-email@example.com
SMTP_PASSWORD=your-email-password
```
